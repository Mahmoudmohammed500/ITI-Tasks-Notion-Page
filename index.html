// Task 1 Code

const video = document.getElementById("myVideo");
const seekBar = document.getElementById("seekBar");
const speedDisplay = document.getElementById("speedDisplay");

function playVideo() {
  video.play();
}

function pauseVideo() {
  video.pause();
}

function stopVideo() {
  video.pause();
  video.currentTime = 0;
}

function changeVolume(val) {
  video.volume = val;
}

function seekVideo(val) {
  video.currentTime = (val / 100) * video.duration;
}

video.addEventListener("timeupdate", () => {
  if (video.duration) {
    seekBar.value = (video.currentTime / video.duration) * 100;
  }
});

function changeSpeed(val) {
  video.playbackRate = parseFloat(val);
  speedDisplay.textContent = val + "x";
}
function rewindVideo() {
  video.currentTime -= 10;
}

function forwardVideo() {
  video.currentTime += 10;
}

function muteVideo() {
  video.muted = !video.muted;
}

function drawHalfCircle() {
  ctx.clearRect(0, 0, canvas.width, canvas.height);

  ctx.beginPath();
  ctx.fillStyle = "yellow";

  if (flipCount >= 10) {
    ctx.arc(400, 250, 100, 0, Math.PI * 2);
  } else {
    if (isTop) {
      ctx.arc(400, 250, 100, Math.PI, 0, false);
    } else {
      ctx.arc(400, 250, 100, 0, Math.PI, false);
    }
    isTop = !isTop;
    flipCount++;
  }

  ctx.fill();
}

function throttel(func, interval) {
  let lastCall = 0;
  return function (...args) {
    const now = Date.now();
    if (now - lastCall >= interval) {
      lastCall = now;
      func.apply(this, args);
    }
  };
}
